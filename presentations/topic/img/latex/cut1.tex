\begin{document}
\newcommand{\term}[3]{
	\node[state] (#1) [#3] {#2};
}%
\newcommand{\termMap}[2]{
	\node[fill=black, draw=black, rectangle, minimum height=0.15em] (#1) [#2] {};
}%
\newcommand{\branch}[2]{
	\draw (#1) -- (#2);
}%
\begin{tikzpicture}[>=stealth',node distance=1em and 0em, auto, semithick, shorten >=0pt, every node/.style={transform shape}]
  \tikzstyle{every state}=[fill=white,align=center,  font=\small, rectangle, minimum height=1.6em, minimum width=1.6em]

\term{T1}{foo (X, Y)}{};

\termMap{M1}{below left=of T1, xshift=-2.5em};
\termMap{M2}{below right=of T1, xshift=2.5em};

\branch{T1.south west}{M1.north east};
\branch{T1.south east}{M2.north west};

\term{T2}{bar (a)}{below left=of M1};
\term{T3}{bar (b)}{below right=of M1};

\branch{M1.south west}{T2.north};
\branch{M1.south east}{T3.north};

\term{T4}{baz (a)}{below left=of M2};
\term{T5}{baz (b)}{below right=of M2};

\branch{M2.south west}{T4.north};
\branch{M2.south east}{T5.north};
 
\end{tikzpicture}
